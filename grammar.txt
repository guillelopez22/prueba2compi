Grammar
=======

input -> stmt_list

stmt_list -> stmt stmt_list'

stmt_list' -> ; stmt ; stmt_list'
	     | lambda

stmt -> assign
        | PRINT expr
        | fun_decl

assign -> IDENT = expr

fun_decl -> IDENT ( arg ) = expr

arg -> NUMBER
      | IDENT

expr -> term expr'

expr' -> + term expr'
	| lambda

term -> factor term'

term' -> * factor term'
	| lambda

factor -> IDENT
        | NUMBER
        | ( expr )
